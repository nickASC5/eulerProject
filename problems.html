Problem 1

Find the sum of all multiples of 3 or 5 below 1000.
sum = 0
for i in range(1000):
	if (i % 3 == 0) or (i % 5 == 0):
		sum += i
print(sum)
=> 233168

Problem 2

By considering the terms in the Fibonacci sequence whose values do not exceed four million, find the sum of the even-valued terms.
fibSeq = [1,2]
while fibSeq[-1] < 4000000:
	fibSeq += [fibSeq[-1] + fibSeq[-2]]

fibSum = 0

for num in fibSeq:
	if (num % 2 == 0):
		fibSum += num

print(fibSum)
=> 4613732

Problem 3

What is the largest prime factor of the number 600851475143?
import math
def isPrime(num):
	for i in range(2,int(math.sqrt(num))+1):
		if (num % i == 0):
			return False
	return True

bigNum = 600851475143
largestPrime = 0
for i in range(1,math.ceil(math.sqrt(bigNum))):
	if (bigNum % i == 0):
		if isPrime(i) and i > largestPrime:
			largestPrime = i
		if isPrime(bigNum / i) and (bigNum / i) > largestPrime:
			largestPrime = bigNum / i

print(largestPrime)
=> 6857
